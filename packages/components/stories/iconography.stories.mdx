import { Meta, IconGallery, IconItem } from '@storybook/addon-docs'
import { Icon } from '../src/Icon'
import * as featherIcons from 'react-feather'

<Meta title="Styles/Iconography" />

export function AllIcons({ icons: allIcons }) {
  return allIcons.map((iconName) => (
    <IconItem name={iconName} key={iconName}>
      <Icon name={iconName} />
    </IconItem>
  ))
}

# Iconography

OpenCRVS UI-Kit is relying on open source [Feather icons](https://github.com/feathericons/feather). In addition to feather icons we have custom ones; record status icons and custom ui-kit icons. The colour of these icons cannot be changed with the `color` -prop.<br />
For more information how to use the [&lt;Icon&gt;](/docs/typography-icon--default) component, see its story page.
